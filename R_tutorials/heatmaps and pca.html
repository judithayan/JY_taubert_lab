Make heatmap

pheatmap(t(data)) # t() is same as pandas df.T

make z score normalization, and adjust other parameters

because we are interested in change relative to row norm z score = how many stds away from row mean. use ?pheatmap to get the full docs

pheatmap(t(data), scale = "row", 
         cluster_rows=TRUE, cluster_cols=FALSE,
         clustering_method = "ward.D2",
         show_rownames=TRUE, show_colnames=FALSE)

use the metadata to annotate the plot

pheatmap(t(data), scale = "row", 
         cluster_rows=TRUE, cluster_cols=FALSE,
         clustering_method = "ward.D2",
         show_rownames=TRUE, show_colnames=FALSE,
         annotation_col = meta)

PCA analysis

you can graph it using ggplot2 too you can also save it down write.table(pca$x, iris.pca.txt)

pca <- prcomp(data, center = TRUE, scale = TRUE) 
plot(pca$x)

## PCAtools package: a simpler way to make PCA

p <- pca(t(data), metadata = meta, center = TRUE, scale = TRUE)

# color has to be a metadata column
biplot(p, colby = "Species", legendPosition = "right", lab = NULL)

SCREE plot to visualize which PC contributes most variation

screeplot(p)

Pairsplot

pairsplot(p, components = getComponents(p, c(1:4)), colby = "Species")
